144    @staticmethod
145    def download_condastat(
146        *data_source: Union[CondaStatDataSource, str],
147        year: str,
148        month: str,
149        day: str,
150        package: str,
151    ) -> Optional[pd.DataFrame]:
152        """
153        A method for downloading and filtering conda statistics.
154
155        Args:
156            *data_source: Data source of the statistics.
157                For example anaconda, conda-forge, bioconda.
158            year: Year of the statistics.
159            month: Month of the statistics.
160            day: Day of the statistics.
161            package: Name of the package.
162
163        Returns:
164            The filtered conda statistics.
165        """
166
167        none_stat = None
168        try:
169            stat_url = "s3://anaconda-package-data/conda/hourly"
170            stat = dd.read_parquet(
171                f"{stat_url}/{year}/{month}/{year}-{month}-{day}.parquet",
172                storage_options={"anon": True},
173            )
174        except FileNotFoundError:
175            return none_stat
176        sources = [CondaStatDataSource(source).value for source in data_source]
177        stat = stat.loc[
178            (stat["data_source"].isin(sources)) & (stat.pkg_name == package)
179        ]
180        stat = stat.drop(columns=["time", "pkg_name"])
181        stat = stat.groupby(
182            ["data_source", "pkg_version", "pkg_platform", "pkg_python"]
183        ).sum()
184        stat = stat.reset_index()
185        stat = stat.compute()
186
187        stat = stat[stat["counts"] != 0]
188        if stat.empty:
189            return none_stat
190        stat.insert(0, "package", f"{package}")
191        stat.insert(0, "date", f"{year}-{month}-{day}")
192        stat.rename(columns={"counts": "downloads"}, inplace=True)
193        return stat

220    def get_condastat(
221        self,
222        *data_source: Union[CondaStatDataSource, str],
223        start_date: Optional[str] = None,
224        end_date: Optional[str] = None,
225    ) -> Optional[pd.DataFrame]:
226        """
227        Returns conda statistics.
228
229        Args:
230            *data_source: Type of the data source.
231                For example CondaStatDataSource.CONDAFORGE.
232            start_date (optional): Start date of the statistics,
233                should be in one of the following formats:
234                - %Y, for example 2022 will be 2022-01-01.
235                - %Y-%m, for example 2022-01 will be 2022-01-01.
236                - %Y-%m-%d, for example 2022-01-01 will be 2022-01-01.
237                - None will be the the date that was two months ago.
238            end_date (optional): End date of the statistics,
239                should be in one of the following formats:
240                - %Y, for example 2022 will be 2022-12-31.
241                - %Y-%m, for example 2022-12 will be 2022-12-31.
242                - %Y-%m-%d, for example 2022-12-31 will be 2022-12-31.
243                - None will be the actual date.
244
245        Returns:
246            The gathered conda statistics.
247        """
248
249        stat_date = StatDate(start=start_date, end=end_date)
250        return self._get_condastat(*data_source, stat_date=stat_date)

252    def _get_condastat(
253        self, *data_source: Union[CondaStatDataSource, str], stat_date: StatDate
254    ) -> Optional[pd.DataFrame]:
255        stats = None
256        for i in range((stat_date.end - stat_date.start).days + 1):
257            date = stat_date.start + timedelta(days=i)
258            year = date.strftime("%Y")
259            month = date.strftime("%m")
260            day = date.strftime("%d")
261            new_stats = WriteCondaStat.download_condastat(
262                *data_source,
263                year=year,
264                month=month,
265                day=day,
266                package=self._package_name,
267            )
268            if new_stats is not None:
269                stats = pd.concat([stats, new_stats], ignore_index=True, sort=False)
270        if stats is not None:
271            if not self.write_package_name:
272                stats.drop(["package"], inplace=True, axis=1, errors="ignore")
273            stats.sort_values(
274                self._get_condastat_columns()[:-1],
275                inplace=True,
276            )
277            date_index = stats.pop("date")
278            stats.insert(0, date_index.name, date_index)
279            stats = WriteCondaStat._set_data_frame_columns(stats)
280        return stats

282    def _get_actual_condastat(
283        self, *data_source: Union[CondaStatDataSource, str]
284    ) -> Optional[pd.DataFrame]:
285        header = ["data_source", "pkg_version", "downloads"]
286        raw_stats = []
287        for source in data_source:
288            json_stats = WriteCondaStat.download_actual_condastat(
289                source, self._package_name
290            )
291            if json_stats is not None:
292                packages = json_stats["files"]
293                for package in packages:
294                    stat = []
295                    stat.append(CondaStatDataSource(source).value)
296                    stat.append(package["version"])
297                    stat.append(package["ndownloads"])
298                    raw_stats.append(stat)
299        stats = pd.DataFrame(raw_stats, columns=header)
300        stats = stats.groupby(["data_source", "pkg_version"]).sum()
301        stats = stats.reset_index()
302        if self.write_package_name:
303            stats.insert(0, "package", f"{self._package_name}")
304        date = datetime.now().strftime("%Y-%m-%d")
305        stats.insert(0, "date", f"{date}")
306        if stats.empty:
307            return None
308        return stats

514    def _concat_with_no_data(
515        self, stat: Optional[pd.DataFrame], stat_date: StatDate, actual: bool
516    ) -> pd.DataFrame:
517        days = WriteCondaStat._get_days(stat_date)
518
519        if actual:
520            no_data = {
521                "date": days,
522                "package": self._package_name,
523                "data_source": np.nan,
524                "pkg_version": np.nan,
525                "downloads": 0,
526            }
527            keys = self._get_actual_condastat_columns()[:-1]
528        else:
529            no_data = {
530                "date": days,
531                "package": self._package_name,
532                "data_source": np.nan,
533                "pkg_version": np.nan,
534                "pkg_platform": np.nan,
535                "pkg_python": np.nan,
536                "downloads": 0,
537            }
538            keys = self._get_condastat_columns()[:-1]
539
540        if not self.write_package_name:
541            del no_data["package"]
542
543        if stat is not None:
544            no_data_df = pd.DataFrame(data=no_data)
545            no_data_df = WriteCondaStat._merge_data_frames(
546                stat,
547                no_data_df,
548                keys,
549            )
550        else:
551            no_data_df = pd.DataFrame(data=no_data)
552            no_data_df = WriteCondaStat._set_data_frame_columns(no_data_df)
553
554        return no_data_df

556    @staticmethod
557    def _merge_data_frames(
558        dataf1: pd.DataFrame, dataf2: pd.DataFrame, keys: List[str]
559    ) -> pd.DataFrame:
560        dataf1 = dataf1.replace("null", np.nan)
561        dataf1 = dataf1.replace("nan", np.nan)
562        dataf1 = WriteCondaStat._set_data_frame_columns(dataf1)
563        dataf2 = dataf2.replace("null", np.nan)
564        dataf2 = dataf2.replace("nan", np.nan)
565        dataf2 = WriteCondaStat._set_data_frame_columns(dataf2)
566        merged = pd.merge(dataf1, dataf2, how="outer", on=[*keys], suffixes=("", "_y"))
567        merged.sort_values([*keys], inplace=True)
568        merged.downloads.fillna(merged.downloads_y, inplace=True)
569        merged.drop(["downloads_y"], inplace=True, axis=1, errors="ignore")
570        return merged

572    @staticmethod
573    def _set_data_frame_columns(dataf: pd.DataFrame) -> pd.DataFrame:
574        for col in dataf.columns:
575            if col != "downloads":
576                dataf[col] = dataf[col].astype(str)
577            else:
578                dataf[col] = dataf[col].astype(int)

581    @staticmethod
582    def _filter_data_frame_rows(stat: pd.DataFrame) -> pd.DataFrame:
583        stat = WriteCondaStat._set_data_frame_columns(stat)
584        stat = stat.drop_duplicates()
585        df_zero = stat[stat["downloads"] == 0]
586        df_no_zero = stat[stat["downloads"] != 0]
587        stat = stat.drop(df_zero.index[df_zero["date"].isin(df_no_zero["date"])])
588        return stat

175        except FileNotFoundError:
176            return none_stat

179            (stat["data_source"].isin(sources)) & (stat.pkg_name == package)

183            ["data_source", "pkg_version", "pkg_platform", "pkg_python"]
184        ).sum()

189        if stat.empty:
190            return none_stat
191        stat.insert(0, "package", f"{package}")
192        stat.insert(0, "date", f"{year}-{month}-{day}")
193        stat.rename(columns={"counts": "downloads"}, inplace=True)
194        return stat
